#ifndef TUPLE_PRINTER_H
#define TUPLE_PRINTER_H

#include <iostream>
#include <utility>
#include <tuple>

namespace meta {
    namespace utility {

        template <typename... Args>
        struct tuple_printer {
            static void print(const std::tuple<Args...>& tuplita, std::ostream& os) {
                std::apply([&os](const Args&... args) {
                  std::size_t indice = 0;
                  ((os << (indice++ ? " " : "") << args), ...);
                }, tuplita);
            }
        };

    }
}

#endif
