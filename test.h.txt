// Vector3D: cumple Addable, Divisible y Comparable → funciona con todos los algoritmos
struct Vector3D {
    double x, y, z;

    Vector3D(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}

    // operator+ → cumple Addable
    Vector3D operator+(const Vector3D& otro) const {
        return {x + otro.x, y + otro.y, z + otro.z};
    }

    // operator- → necesario para variance (d = valor - promedio)
    Vector3D operator-(const Vector3D& otro) const {
        return {x - otro.x, y - otro.y, z - otro.z};
    }

    // operator* → necesario para variance (d * d)
    Vector3D operator*(const Vector3D& otro) const {
        return {x * otro.x, y * otro.y, z * otro.z};
    }

    // operator/ size_t → cumple Divisible
    Vector3D operator/(std::size_t n) const {
        return {x / n, y / n, z / n};
    }

    // operator< y operator> → cumple Comparable (compara módulo)
    bool operator<(const Vector3D& otro) const {
        return (x*x + y*y + z*z) < (otro.x*otro.x + otro.y*otro.y + otro.z*otro.z);
    }
    bool operator>(const Vector3D& otro) const { return otro < *this; }

    friend std::ostream& operator<<(std::ostream& os, const Vector3D& v) {
        return os << "(" << v.x << ", " << v.y << ", " << v.z << ")";
    }
};

// Coordenada: solo tiene operator+ → cumple Addable pero NO Divisible ni Comparable
// Por eso falla en mean, variance y max.
struct Coordenada {
    double lat, lon;

    Coordenada(double lat = 0, double lon = 0) : lat(lat), lon(lon) {}

    // operator+ → cumple Addable
    Coordenada operator+(const Coordenada& otro) const {
        return {lat + otro.lat, lon + otro.lon};
    }

    // NO tiene operator/ → NO cumple Divisible
    // NO tiene operator< → NO cumple Comparable

    friend std::ostream& operator<<(std::ostream& os, const Coordenada& c) {
        return os << "[" << c.lat << ", " << c.lon << "]";
    }
};
